// React Functional Component

[<ReactComponent>]
let Welcome (props: {| name: string |}) =
  Html.h1 $"Hello, {props.name}"

let element = Welcome {| name = "Alan" |}

ReactDom.render(
  element,
  document.getElementById("root"))
